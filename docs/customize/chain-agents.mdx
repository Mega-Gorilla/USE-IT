```python
from browser_use import Agent, Browser

# Create browser and keep it alive
browser = Browser(
	headless=False,
	keep_alive=True,  # Don't close browser after each agent
)
tasks = [
	'Search for Browser Use',
	'Click on the first link',
	'Find the social media page',
]

async def main():
	for task in tasks:
		agent = Agent(task=task, browser=browser)
		await agent.run()


	await browser.stop()

```
